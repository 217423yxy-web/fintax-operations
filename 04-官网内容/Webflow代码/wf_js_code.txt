function buildCard(ev) {
  if (ev.type === 'photo') {
    return `<div class="ev-card ${ev.size}" style="position:relative;overflow:hidden;">
      <img src="${ev.src}" alt="${ev.label}"
        style="width:100%;height:100%;object-fit:cover;display:block;">
      <div style="position:absolute;bottom:0;left:0;right:0;
                  padding:8px 12px;
                  background:linear-gradient(transparent,rgba(0,0,0,0.60));
                  color:#fff;">
        <div style="font-size:12px;font-weight:700;letter-spacing:-0.01em;">${ev.label}</div>
        <div style="font-size:10px;opacity:0.65;margin-top:2px;">${ev.date}</div>
      </div>
    </div>`;
  }
  const name = ev.name.replace('\n','<br>');
  return `<div class="ev-card ${ev.theme} ${ev.size}">
    <div class="ev-inner">
      <div class="ev-top">
        <span class="ev-tag">${ev.tag}</span>
        <span class="ev-date">${ev.date}</span>
      </div>
      <div class="ev-bottom">
        <div class="ev-name">${name}</div>
        <div class="ev-org">${ev.org}</div>
      </div>
    </div>
    <div class="ev-num">${ev.num}</div>
  </div>`;
}

function fillRow(id, data) {
  const el = document.getElementById(id);
  const html = [...data, ...data].map(buildCard).join('');
  el.innerHTML = html;
}

const CFG={
  dash:{url:'app.fintax.tech / dashboard', sn:'sn-dash'},
  acct:{url:'app.fintax.tech / accounting',sn:'sn-acct'},
  calc:{url:'app.fintax.tech / tax-calc',  sn:'sn-calc'},
  rep: {url:'app.fintax.tech / reports',   sn:'sn-rep'},
};
let cur='dash';
function go(id){
  if(id===cur)return;
  const old=document.getElementById('s-'+cur);
  old.classList.remove('show');
  setTimeout(()=>{old.classList.remove('on');old.style.display='none';},240);
  const nxt=document.getElementById('s-'+id);
  nxt.style.display='flex';nxt.classList.add('on');
  requestAnimationFrame(()=>requestAnimationFrame(()=>nxt.classList.add('show')));
  document.querySelectorAll('.mtab').forEach(t=>t.classList.toggle('active',t.dataset.s===id));
  Object.keys(CFG).forEach(k=>{const el=document.getElementById(CFG[k].sn);if(el)el.classList.toggle('active',k===id);});
  document.getElementById('urlBar').textContent=CFG[id]?.url??'';
  cur=id;
}
document.querySelectorAll('.mtab').forEach(t=>t.addEventListener('click',()=>go(t.dataset.s)));
Object.keys(CFG).forEach(k=>{const el=document.getElementById(CFG[k].sn);if(el)el.addEventListener('click',()=>go(k));});
